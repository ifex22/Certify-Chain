<script>
const student = document.getElementById("student");
const course = document.getElementById("course");
const issuer = document.getElementById("issuer");
const dateF = document.getElementById("date");
const notes = document.getElementById("notes");

student.addEventListener("input",()=>document.getElementById("prev-student").textContent = student.value || "Student Name");
course.addEventListener("input",()=>document.getElementById("prev-course").textContent = course.value || "Course Title");
issuer.addEventListener("input",()=>document.querySelector("#prev-issuer span").textContent = issuer.value || "Organization");
dateF.addEventListener("input",()=>document.querySelector("#prev-date span").textContent = dateF.value || "YYYY-MM-DD");
notes.addEventListener("input",()=>document.getElementById("prev-notes").textContent = notes.value || "Notes will appear here…");

async function downloadPNG() {
  const node = document.getElementById("certificate");
  const canvas = await html2canvas(node);
  const link = document.createElement("a");
  link.download = "certificate.png";
  link.href = canvas.toDataURL();
  link.click();
}

async function downloadPDF() {
  const node = document.getElementById("certificate");
  const canvas = await html2canvas(node);
  const img = canvas.toDataURL("image/png");

  const pdf = new jspdf.jsPDF("landscape", "px", "a4");
  pdf.addImage(img, "PNG", 20, 20, 800, 550);
  pdf.save("certificate.pdf");
}
</script>
